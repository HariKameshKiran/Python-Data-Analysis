# Ensure 'DATE_RECEIVED' is in datetime format
if 'DATE_RECEIVED' in df.columns:
    df['DATE_RECEIVED'] = pd.to_datetime(df['DATE_RECEIVED'], errors='coerce')
    df['YearMonth'] = df['DATE_RECEIVED'].dt.to_period('M')

# Group by brand and date to identify spikes
brand_trends = df.groupby(['BRAND_NAME', 'YearMonth']).size().reset_index(name='Report_Count')

# Plot time series for top 3 most reported brands
top_3_brands = df['BRAND_NAME'].value_counts().head(3).index.tolist()
for brand in top_3_brands:
    subset = brand_trends[brand_trends['BRAND_NAME'] == brand]
    subset.set_index('YearMonth')['Report_Count'].plot(figsize=(10, 4), title=f'Report Trend for {brand}')
    plt.ylabel('Reports')
    plt.xlabel('Month')
    plt.xticks(rotation=45)
